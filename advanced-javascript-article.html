01 — The Foundation

Closures & Lexical Scope

A closure is a function that remembers its lexical environment even when executed outside of it. This is arguably the most important concept in JavaScript.

Every function in JS forms a closure over the variables in scope at the time it was defined. The inner function maintains a live reference — not a snapshot — to those variables.

JavaScript — Closure Fundamentals 
// ── Basic closure ────────────────────────────────
function makeCounter(start = 0) {
  let count = start  // closed over by inner functions

  return {
    increment: () => ++count,
    decrement: () => --count,
    reset:     () => { count = start },
    value:     () => count,
  }
}

const counter = makeCounter(10)
counter.increment()  // 11
counter.increment()  // 12
counter.reset()      // back to 10

// ── Private state via closure ────────────────────
function createWallet(initialBalance) {
  let balance = initialBalance   // truly private
  const history = []

  return {
    deposit(amount) {
      balance += amount
      history.push({ type: "deposit", amount, balance })
    },
    getBalance() { return balance },
    getHistory() { return [...history] },  // copy, not reference
  }
}

The Classic Loop Closure Bug

for (var i = 0; i < 3; i++) {

// Memoization using closure
function memoize(fn) {
  const cache = new Map()
  return function(...args) {
    const key = JSON.stringify(args)
    if (cache.has(key)) return cache.get(key)
    const result = fn.apply(this, args)
    cache.set(key, result)
    return result
  }
}

const fib = memoize(function fibonacci(n) {
  return n <= 1 ? n : fib(n - 1) + fib(n - 2)
})
02 — The Object System

Prototypes, Inheritance & the Prototype Chain

JavaScript is a prototype-based language. Every object has an internal [[Prototype]] link — classes in JS are syntactic sugar over this mechanism.

JavaScript — Prototype Chain 
// ── How prototype lookup works ───────────────────
const animal = {
  breathe() { return "breathing" }
}
const dog = Object.create(animal)
dog.bark = function() { return "woof" }

dog.bark()    // found on dog directly
dog.breathe() // walks up chain to animal

// dog → animal → Object.prototype → null
Object.getPrototypeOf(dog) === animal  // true

// ── Constructor function pattern ─────────────────
function Person(name, age) {
  this.name = name
  this.age  = age
}
// Shared method — only one copy in memory
Person.prototype.greet = function() {
  return `Hi, I'm ${this.name}`
}

const alice = new Person("Alice", 30)
alice.greet()            // "Hi, I'm Alice"
alice instanceof Person  // true

// ── ES6 classes = syntactic sugar ────────────────
class Vehicle {
  #fuelLevel = 100  // true private field

  constructor(make, model) {
    this.make  = make
    this.model = model
  }

  get fuel()     { return this.#fuelLevel }
  refuel(amount) { this.#fuelLevel = Math.min(100, this.#fuelLevel + amount) }
  toString()    { return `${this.make} ${this.model}` }
}

class ElectricCar extends Vehicle {
  #batteryLevel = 80

  constructor(make, model, range) {
    super(make, model)
    this.range = range
  }

  static compare(a, b) {
    return a.range - b.range
  }
}
💡 How new works
When you call new Fn(), JS creates a new object whose [[Prototype]] is Fn.prototype, runs Fn with this pointing to that object, and returns it (unless the constructor returns a different object explicitly).
03 — Asynchronous JS

Async Patterns: Callbacks → Promises → Async/Await

JavaScript is single-threaded but non-blocking. Understanding how it handles async operations is essential for writing performant apps.

Promises — The Full API

JavaScript 
// ── Creating and chaining ────────────────────────
const delay = (ms) => new Promise(resolve => setTimeout(resolve, ms))

fetchUser(1)
  .then(user    => fetchPosts(user.id))
  .then(posts   => posts.filter(p => p.published))
  .catch(err    => console.error(err))
  .finally(()  => setLoading(false))

// ── Promise combinators ──────────────────────────
const [users, posts, comments] = await Promise.all([
  fetchUsers(),
  fetchPosts(),
  fetchComments(),
])

// allSettled — never rejects, gives status for each
const results = await Promise.allSettled([p1, p2, p3])
results.forEach(r => {
  if (r.status === "fulfilled") console.log(r.value)
  else                            console.error(r.reason)
})

Promise.race([p1, p2])  // first to settle wins
Promise.any([p1, p2])   // first to FULFILL wins

Async/Await Patterns

JavaScript 
// ── Sequential vs parallel ───────────────────────
async function sequential() {
  const a = await fetchA()   // waits for A
  const b = await fetchB()   // then waits for B = slow!
}

async function parallel() {
  const [a, b] = await Promise.all([fetchA(), fetchB()])  // both at once
}

// ── Retry with exponential backoff ───────────────
async function retry(fn, maxAttempts = 3, baseDelay = 300) {
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    try {
      return await fn()
    } catch (err) {
      if (attempt === maxAttempts) throw err
      await delay(baseDelay * 2 ** (attempt - 1))  // 300, 600, 1200ms
    }
  }
}

// ── AbortController for cancellable fetches ──────
async function fetchWithTimeout(url, ms = 5000) {
  const controller = new AbortController()
  const id = setTimeout(() => controller.abort(), ms)
  try {
    const res = await fetch(url, { signal: controller.signal })
    return await res.json()
  } finally {
    clearTimeout(id)
  }
}
04 — Concurrency Model

The Event Loop, Call Stack & Task Queues

JavaScript has a single call stack, but it never blocks. Understanding the event loop is the key to understanding why async JS behaves the way it does.

How the Event Loop Works
The JS runtime has a call stack (LIFO), a heap (memory), and a message queue. The event loop continuously checks: if the stack is empty, pull the next message from the queue and push it onto the stack.
  • 1
    Call Stack — synchronous code executes here. Only one frame at a time. If it blocks, everything blocks.
  • 2
    Microtask Queue — Promise callbacks (.then, .catch), queueMicrotask(), MutationObserver. Drained completely before the next macrotask.
  • 3
    Macrotask Queue — setTimeout, setInterval, I/O callbacks, requestAnimationFrame. One runs per event loop iteration.
JavaScript — Execution Order 
console.log("1 — sync")

setTimeout(() => console.log("4 — macrotask"), 0)

Promise.resolve()
  .then(() => console.log("3 — microtask"))

console.log("2 — sync")

// Output: 1, 2, 3, 4
// Sync first → microtasks (Promises) → macrotasks (setTimeout)

// ── queueMicrotask ───────────────────────────────
queueMicrotask(() => {
  // Runs at end of current task, before next macrotask
  // Useful for deferring work without full async overhead
})

// ── Starving the event loop (bad) ───────────────
function blockingLoop() {
  const end = Date.now() + 2000
  while (Date.now() < end) {}  // freezes the browser for 2s!
}

// ── Non-blocking chunk processing (good) ────────
async function processInChunks(items, chunkSize = 100) {
  for (let i = 0; i < items.length; i += chunkSize) {
    processChunk(items.slice(i, i + chunkSize))
    await new Promise(r => setTimeout(r))  // yield to event loop
  }
}
05 — Functional Programming

Functional Programming Techniques

FP in JS isn't dogma — it's a toolkit. Pure functions, immutability, composition, and higher-order functions lead to more predictable, testable code.

JavaScript 
// ── Currying ─────────────────────────────────────
const curry = (fn) => {
  const arity = fn.length
  return function curried(...args) {
    return args.length >= arity
      ? fn(...args)
      : (...more) => curried(...args, ...more)
  }
}

const add    = curry((a, b) => a + b)
const add5   = add(5)          // partially applied
add5(3)                        // 8

// ── Function composition ─────────────────────────
const compose = (...fns) => (x) => fns.reduceRight((v, f) => f(v), x)
const pipe    = (...fns) => (x) => fns.reduce((v, f) => f(v), x)

const double   = x => x * 2
const addOne   = x => x + 1
const toString = x => `Result: ${x}`

const transform = pipe(double, addOne, toString)
transform(5)  // "Result: 11"  →  5*2=10, +1=11

// ── Transducer pattern ───────────────────────────
// Composes map/filter without creating intermediate arrays
const xform = compose(
  map(x => x * 2),
  filter(x => x % 3 === 0)
)

// ── Immutable update patterns ────────────────────
const user    = { name: "Alice", age: 30, prefs: { theme: "dark" } }
const updated = { ...user, age: 31, prefs: { ...user.prefs, theme: "light" } }
06 — The this Keyword

this, bind, call & apply

this is not about where a function is defined — it's about how it is called. It's the most misunderstood part of JavaScript.

Call style this value
fn() — plain call undefined (strict) or window
obj.method() — method call obj
new Fn() — constructor newly created object
fn.call(ctx, a, b) ctx
fn.apply(ctx, [a, b]) ctx
fn.bind(ctx) ctx (permanent)
() => {} — arrow function lexical (outer this)
JavaScript 
// ── Method context ────────────────────────────────
const timer = {
  seconds: 0,

  // ❌ Regular function — loses `this` in callback
  startBad() {
    setInterval(function() {
      this.seconds++  // `this` is undefined / window
    }, 1000)
  },

  // ✅ Arrow function — inherits `this` from startGood
  startGood() {
    setInterval(() => {
      this.seconds++  // `this` is timer
    }, 1000)
  }
}

// ── call / apply / bind ───────────────────────────
function introduce(greeting, punctuation) {
  return `${greeting}, I'm ${this.name}${punctuation}`
}

const alice = { name: "Alice" }
introduce.call(alice, "Hello", "!")   // "Hello, I'm Alice!"
introduce.apply(alice, ["Hey", "."])  // "Hey, I'm Alice."

const aliceIntro = introduce.bind(alice, "Hi")
aliceIntro("!")  // "Hi, I'm Alice!" — bound forever
07 — Design Patterns

Essential JavaScript Design Patterns

JavaScript — Observer / EventEmitter 
class EventEmitter {
  #events = new Map()

  on(event, listener) {
    if (!this.#events.has(event))
      this.#events.set(event, new Set())
    this.#events.get(event).add(listener)
    return () => this.off(event, listener)  // unsubscribe fn
  }

  off(event, listener) {
    this.#events.get(event)?.delete(listener)
  }

  emit(event, ...args) {
    this.#events.get(event)?.forEach(fn => fn(...args))
  }

  once(event, listener) {
    const wrapper = (...args) => {
      listener(...args)
      this.off(event, wrapper)
    }
    return this.on(event, wrapper)
  }
}
JavaScript — Singleton, Factory & Proxy 
// ── Singleton ─────────────────────────────────────
class Config {
  static #instance = null
  #settings = {}

  static getInstance() {
    return (Config.#instance ??= new Config())
  }
  get(key)      { return this.#settings[key] }
  set(key, val) { this.#settings[key] = val }
}

// ── Factory ───────────────────────────────────────
function createLogger(level = "info") {
  const prefix = `[${level.toUpperCase()}]`
  return {
    log(msg)   { console.log(prefix, msg) },
    error(msg) { console.error(prefix, msg) },
  }
}

// ── Proxy — intercept object operations ──────────
function createValidator(target, schema) {
  return new Proxy(target, {
    set(obj, prop, value) {
      const validator = schema[prop]
      if (validator && !validator(value)) {
        throw new TypeError(`Invalid value for ${prop}`)
      }
      obj[prop] = value
      return true
    }
  })
}

const user = createValidator({}, {
  age: v => typeof v === "number" && v > 0
})
user.age = 25   // ✓
user.age = -1   // ✗ TypeError

08 — Under the Hood

JS Engine Internals & V8

Understanding V8 helps you write code the engine can optimise. Small changes in code patterns can have dramatic performance effects.

⚙️

Parsing & AST

V8 parses your source into an Abstract Syntax Tree, then uses Ignition (bytecode interpreter) before optimizing hot paths with TurboFan.

🔥

JIT Compilation

Functions called repeatedly are JIT-compiled to native machine code. V8 tracks type feedback — changing types causes deoptimization.

🗺️

Hidden Classes

V8 creates hidden classes for object shapes. Objects with the same properties in the same order share a hidden class — which is fast.

♻️

Garbage Collection

V8 uses generational GC: Scavenger for the young generation, Mark-Compact for the old generation. Short-lived objects are cheap.

JavaScript — V8 Optimisation Tips 
// ── Hidden classes — keep property order consistent
function Point(x, y) {
  this.x = x  // always x first
  this.y = y  // then y
}
// Good: new Point(1,2) and new Point(3,4) share hidden class
// Bad: adding properties conditionally in different orders

// ── Monomorphic vs polymorphic functions ─────────
// V8 optimizes when a function always receives the same types
function addFast(a, b) { return a + b }
addFast(1, 2)      // V8 specializes for numbers
addFast("a", "b")  // type change → deoptimize! 😱

// ── Avoid polymorphic hidden class transitions ───
const obj = {}
obj.x = 1  // shape A
obj.y = 2  // shape B — forces transition
// Better: obj = { x: 1, y: 2 } — initialized in one step

// ── Array type stability ─────────────────────────
const nums = [1, 2, 3, 4]   // SMI (small integer) array — fast!
nums.push(3.14)             // becomes DOUBLE array
nums.push("oops")           // becomes GENERIC array — slow
09 — Memory

Memory Management & Leak Prevention

JavaScript — Common Memory Leaks 
// ── Leak 1: Forgotten event listeners ────────────
class Component {
  constructor() {
    this.handler = this.onResize.bind(this)
    window.addEventListener("resize", this.handler)
  }
  destroy() {
    window.removeEventListener("resize", this.handler)  // ← clean up!
  }
}

// ── Leak 2: Closures holding large data ──────────
function processHugeArray() {
  const bigArray = new Array(1_000_000).fill(0)
  const result = bigArray.length

  // Only capture what you need, not the whole array
  return () => result  // ✓ bigArray can be GCed
}

// ── WeakMap / WeakRef for non-leaking caches ─────
const cache = new WeakMap()  // keys GCed when no other references

function expensive(obj) {
  if (cache.has(obj)) return cache.get(obj)
  const result = compute(obj)
  cache.set(obj, result)
  return result
}

// ── WeakRef — hold object without preventing GC ──
const ref  = new WeakRef(someObject)
const obj  = ref.deref()
if (obj) {
  // object still exists
}

Generators & Iterators

JavaScript 
// ── Generator functions ──────────────────────────
function* range(start, end, step = 1) {
  for (let i = start; i < end; i += step) yield i
}
[...range(0, 10, 2)]  // [0, 2, 4, 6, 8]

// ── Infinite generator ───────────────────────────
function* fibonacci() {
  let [a, b] = [0, 1]
  while (true) {
    yield a;
    [a, b] = [b, a + b]
  }
}
const fib = fibonacci()
Array.from({ length: 8 }, () => fib.next().value)
// [0, 1, 1, 2, 3, 5, 8, 13]

// ── Custom iterable ──────────────────────────────
class Range {
  constructor(from, to) {
    this.from = from; this.to = to
  }
  [Symbol.iterator]*() {
    for (let i = this.from; i <= this.to; i++) yield i
  }
}
[...new Range(1, 5)]  // [1, 2, 3, 4, 5]

12 — Performance

Performance Patterns & Best Practices

JavaScript — Debounce & Throttle 
// ── Debounce — delay until quiet ─────────────────
function debounce(fn, wait) {
  let timer
  return function(...args) {
    clearTimeout(timer)
    timer = setTimeout(() => fn.apply(this, args), wait)
  }
}

// ── Throttle — at most once per interval ─────────
function throttle(fn, limit) {
  let inThrottle
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }
}

const onSearch   = debounce(search, 300)    // search input
const onScroll   = throttle(handleScroll, 16)  // ~60fps

// ── requestAnimationFrame for animations ─────────
function animate(update) {
  let rafId
  const loop = (timestamp) => {
    update(timestamp)
    rafId = requestAnimationFrame(loop)
  }
  rafId = requestAnimationFrame(loop)
  return () => cancelAnimationFrame(rafId)  // cleanup
}

// ── Structural cloning ───────────────────────────
const deepCopy = structuredClone(original)  // fast native deep clone
